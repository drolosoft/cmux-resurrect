# crex v1.5.0 — Setup Wizard, Watch Daemon, TUI Launcher

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Ship three features that close competitive UX gaps before the r/Ghostty launch: a first-run wizard (`crex setup`), a background watch daemon (`crex watch --daemon`), and an interactive TUI launcher (`crex tui`).

**Architecture:** Each feature is a new Cobra command in `cmd/` with business logic in `internal/`. The wizard uses the existing `huh` library for interactive prompts. The daemon adds PID-file management and shell-hook generation to the existing `orchestrate.Watcher`. The TUI uses Bubble Tea (already an indirect dependency via `huh`) for a fuzzy-search launcher. All three features are independent — each can be built, tested, and committed without the others.

**Tech Stack:** Go 1.26, Cobra CLI, charmbracelet/huh (forms), charmbracelet/bubbletea + bubbles + lipgloss (TUI), TOML config, table-driven tests.

// WriteConfigIfNotExists writes a default config file if one doesn't exist.
// Returns (true, nil) if the file was created, (false, nil) if it already exists.
func WriteConfigIfNotExists(path, watchInterval string, maxAutosaves int) (bool, error) {
	if _, err := os.Stat(path); err == nil {
		return false, nil
	}

	if err := os.MkdirAll(filepath.Dir(path), 0o755); err != nil {
		return false, fmt.Errorf("create config dir: %w", err)
	}

	content := GenerateDefaultConfig(watchInterval, maxAutosaves)
	if err := os.WriteFile(path, []byte(content), 0o644); err != nil {
		return false, fmt.Errorf("write config: %w", err)
	}
	return true, nil
}

func runSetup(cmd *cobra.Command, args []string) error {
	o := newWF(os.Stderr)

	// Step 1: Backend detection
	o.ln()
	o.ln(headingStyle.Render("  Step 1/4 — Backend Detection"))
	o.ln()

	detected := client.Detect()
	backendName := setup.DescribeBackend(detected)

	switch detected {

		o.f("  %s  %s\n", greenStyle.Render("✓"), fmt.Sprintf("cmux detected (env: CMUX_SOCKET_PATH)"))

		o.f("  %s  %s\n", greenStyle.Render("✓"), fmt.Sprintf("Ghostty detected (running)"))

		o.f("  %s  %s\n", yellowStyle.Render("?"), "No terminal backend detected")
		o.f("  %s\n", dimStyle.Render("  crex works with cmux and Ghostty. Launch from inside one of them."))
	}
	_ = backendName

	// Step 2: Config creation
	o.ln()
	o.ln(headingStyle.Render("  Step 2/4 — Configuration"))
	o.ln()

	configPath := config.DefaultConfigPath()
	created, err := setup.WriteConfigIfNotExists(configPath, "5m", 10)
	if err != nil {
		return fmt.Errorf("config: %w", err)
	}

	if created {
		o.f("  %s  %s\n", greenStyle.Render("✓"), fmt.Sprintf("Created %s", configPath))
	} else {
		o.f("  %s  %s\n", dimStyle.Render("–"), fmt.Sprintf("Config already exists at %s", configPath))
	}

	// Step 3: Layouts directory
	o.ln()
	o.ln(headingStyle.Render("  Step 3/4 — Layouts Directory"))
	o.ln()

	layoutsDir := config.DefaultLayoutsDir()
	if err := os.MkdirAll(layoutsDir, 0o755); err != nil {
		return fmt.Errorf("create layouts dir: %w", err)
	}
	o.f("  %s  %s\n", greenStyle.Render("✓"), fmt.Sprintf("Ready at %s", layoutsDir))

	// Step 4: First save (only if a backend is available)
	o.ln()
	o.ln(headingStyle.Render("  Step 4/4 — First Save"))
	o.ln()

	if detected == client.BackendUnknown {
		o.f("  %s  %s\n", dimStyle.Render("–"), "Skipped (no backend detected)")
	} else if setupDefaults {
		if err := doFirstSave(o, detected); err != nil {
			o.f("  %s  %s\n", yellowStyle.Render("!"), fmt.Sprintf("Save failed: %v", err))
		}
	} else {
		o.f("  %s\n", dimStyle.Render("Run 'crex save my-layout' to save your current workspace layout."))
	}

	// Summary
	o.ln()
	o.ln(greenStyle.Render("  ✅ Setup complete!"))
	o.ln()
	o.f("  %s\n", dimStyle.Render("Quick start:"))
	o.f("    %s  %s\n", cyanStyle.Render("crex save my-day"), dimStyle.Render("save current layout"))
	o.f("    %s  %s\n", cyanStyle.Render("crex restore my-day"), dimStyle.Render("restore it later"))
	o.f("    %s  %s\n", cyanStyle.Render("crex watch"), dimStyle.Render("auto-save periodically"))
	o.f("    %s  %s\n", cyanStyle.Render("crex template list"), dimStyle.Render("browse templates"))
	o.ln()

func doFirstSave(o wf, detected client.DetectedBackend) error {
	var cl client.Backend
	switch detected {
	case client.BackendGhostty:
		cl = client.NewGhosttyClient()
	default:
		cl = client.NewCLIClient()
	}

	store, err := newStore()
	if err != nil {
		return err
	}

	saver := &orchestrate.Saver{Client: cl, Store: store}
	layout, err := saver.Save("initial", "created by crex setup")
	if err != nil {
		return err
	}

	o.f("  %s  %s\n", greenStyle.Render("✓"),
		fmt.Sprintf("Saved %d workspaces as 'initial'", len(layout.Workspaces)))
	return nil
}

// internal/orchestrate/daemon.go
package orchestrate

import (
	"fmt"
	"os"
	"path/filepath"
	"strconv"
	"strings"
	"syscall"
)

// DefaultPIDPath returns ~/.config/crex/crex.pid.
func DefaultPIDPath() string {
	home, _ := os.UserHomeDir()
	return filepath.Join(home, ".config", "crex", "crex.pid")
}

// DefaultLogPath returns ~/.config/crex/watch.log.
func DefaultLogPath() string {
	home, _ := os.UserHomeDir()
	return filepath.Join(home, ".config", "crex", "watch.log")
}

// WritePIDFile writes a PID to the given file path.
func WritePIDFile(path string, pid int) error {
	if err := os.MkdirAll(filepath.Dir(path), 0o755); err != nil {
		return fmt.Errorf("create pid dir: %w", err)
	}
	return os.WriteFile(path, []byte(strconv.Itoa(pid)), 0o644)
}

// ReadPIDFile reads a PID from the given file path.
func ReadPIDFile(path string) (int, error) {
	data, err := os.ReadFile(path)
	if err != nil {
		return 0, err
	}
	pid, err := strconv.Atoi(strings.TrimSpace(string(data)))
	if err != nil {
		return 0, fmt.Errorf("invalid pid in %s: %w", path, err)
	}
	return pid, nil
}

// RemovePIDFile deletes the PID file.
func RemovePIDFile(path string) {
	os.Remove(path)
}

// IsDaemonRunning checks if a daemon process is alive by reading the PID file
// and sending signal 0 to check process existence.
func IsDaemonRunning(pidPath string) (bool, int) {
	pid, err := ReadPIDFile(pidPath)
	if err != nil {
		return false, 0
	}

	process, err := os.FindProcess(pid)
	if err != nil {
		return false, pid
	}

	// Signal 0 checks if process exists without actually signaling it.
	err = process.Signal(syscall.Signal(0))
	if err != nil {
		return false, pid
	}
	return true, pid
}
```

// OpenLogWriter opens a log file for writing, rotating if it exceeds maxBytes.
// The old log is renamed to path + ".old" (only one backup is kept).
func OpenLogWriter(path string, maxBytes int64) (*os.File, error) {
	if err := os.MkdirAll(filepath.Dir(path), 0o755); err != nil {
		return nil, fmt.Errorf("create log dir: %w", err)
	}

	// Check if rotation is needed.
	if info, err := os.Stat(path); err == nil && info.Size() >= maxBytes {
		oldPath := path + ".old"
		os.Remove(oldPath)
		os.Rename(path, oldPath)
	}

	return os.OpenFile(path, os.O_CREATE|os.O_WRONLY|os.O_APPEND, 0o644)
}
```

func DetectShell() string {
	shell := os.Getenv("SHELL")
	if shell == "" {
		return ""
	}
	return filepath.Base(shell)
}

// ShellHook returns a shell snippet that auto-starts crex watch on terminal launch.
// The snippet is designed to be added to .zshrc, .bashrc, or config.fish.
// Returns empty string for unsupported shells.
func ShellHook(shell string) string {
	pidPath := DefaultPIDPath()

	switch strings.ToLower(shell) {
	case "zsh", "bash":
		return posixHook(pidPath)
	case "fish":
		return fishHook(pidPath)
	default:
		return ""
	}
}

func posixHook(pidPath string) string {
	return `# crex auto-persistence — add to .zshrc or .bashrc
# Starts crex watch in daemon mode if not already running.
if [ -z "$CREX_NO_WATCH" ]; then
  if ! kill -0 "$(cat ` + `"` + pidPath + `"` + ` 2>/dev/null)" 2>/dev/null; then
    crex watch --daemon &>/dev/null
  fi
fi
`
}

func fishHook(pidPath string) string {
	return `# crex auto-persistence — add to ~/.config/fish/config.fish
# Starts crex watch in daemon mode if not already running.
if not set -q CREX_NO_WATCH
  if not kill -0 (cat "` + pidPath + `" 2>/dev/null) 2>/dev/null
    crex watch --daemon &>/dev/null
  end
end
`
}

// internal/tui/model.go
package tui

import (
	"strings"

	"github.com/charmbracelet/bubbles/textinput"
	tea "github.com/charmbracelet/bubbletea"
)

type modelState int

const (
	stateList   modelState = iota
	stateFilter
)

// SelectedAction is returned when the user selects an item.
type SelectedAction struct {
	Item Item
}

// SaveAction is returned when the user presses 's' to save current layout.
type SaveAction struct{}

// DeleteAction is returned when the user presses 'd' to delete a layout.
type DeleteAction struct {
	Item Item
}

// Model is the Bubble Tea model for the crex TUI launcher.
type Model struct {
	allItems     []Item
	filtered     []Item
	cursor       int
	state        modelState
	keys         KeyMap
	filterInput  textinput.Model
	width        int
	height       int
	selected     *SelectedAction
	action       interface{} // SelectedAction, SaveAction, or DeleteAction
	quitting     bool
}

// NewModel creates a new TUI model with the given layout and template items.
func NewModel(layouts []Item, templates []Item) Model {
	ti := textinput.New()
	ti.Placeholder = "Type to filter..."
	ti.CharLimit = 64

	all := make([]Item, 0, len(layouts)+len(templates))
	all = append(all, layouts...)
	all = append(all, templates...)

	return Model{
		allItems:    all,
		filtered:    all,
		keys:        DefaultKeyMap(),
		filterInput: ti,
	}
}

// Init implements tea.Model.
func (m Model) Init() tea.Cmd {
	return nil
}

// Update implements tea.Model.
func (m Model) Update(msg tea.Msg) (tea.Model, tea.Cmd) {
	switch msg := msg.(type) {
	case tea.WindowSizeMsg:
		m.width = msg.Width
		m.height = msg.Height
		return m, nil

	case tea.KeyMsg:
		return m.handleKey(msg)
	}

	if m.state == stateFilter {
		var cmd tea.Cmd
		m.filterInput, cmd = m.filterInput.Update(msg)
		m.applyFilter()
		return m, cmd
	}

	return m, nil
}

func (m Model) handleKey(msg tea.KeyMsg) (tea.Model, tea.Cmd) {
	// In filter mode, handle special keys first.
	if m.state == stateFilter {
		switch {
		case msg.Type == tea.KeyEsc:
			m.state = stateList
			m.filterInput.SetValue("")
			m.filtered = m.allItems
			m.filterInput.Blur()
			return m, nil
		case msg.Type == tea.KeyEnter:
			// Select the highlighted item.
			if len(m.filtered) > 0 && m.cursor < len(m.filtered) {
				m.action = SelectedAction{Item: m.filtered[m.cursor]}
				m.quitting = true
				return m, tea.Quit
			}
			return m, nil
		default:
			var cmd tea.Cmd
			m.filterInput, cmd = m.filterInput.Update(msg)
			m.applyFilter()
			return m, cmd
		}
	}

	// List mode key handling.
	switch {
	case msg.String() == "q" || msg.Type == tea.KeyCtrlC:
		m.quitting = true
		return m, tea.Quit

	case msg.String() == "/":
		m.state = stateFilter
		m.filterInput.Focus()
		return m, textinput.Blink

	case msg.Type == tea.KeyEnter:
		if len(m.filtered) > 0 && m.cursor < len(m.filtered) {
			m.action = SelectedAction{Item: m.filtered[m.cursor]}
			m.quitting = true
			return m, tea.Quit
		}
		return m, nil

	case msg.String() == "s":
		m.action = SaveAction{}
		m.quitting = true
		return m, tea.Quit

	case msg.String() == "d":
		if len(m.filtered) > 0 && m.cursor < len(m.filtered) {
			item := m.filtered[m.cursor]
			if item.Kind == KindLayout {
				m.action = DeleteAction{Item: item}
				m.quitting = true
				return m, tea.Quit
			}
		}
		return m, nil

	case msg.Type == tea.KeyUp || msg.String() == "k":
		if m.cursor > 0 {
			m.cursor--
		}
		return m, nil

	case msg.Type == tea.KeyDown || msg.String() == "j":
		if m.cursor < len(m.filtered)-1 {
			m.cursor++
		}
		return m, nil
	}

	return m, nil
}

func (m *Model) applyFilter() {
	query := strings.ToLower(m.filterInput.Value())
	if query == "" {
		m.filtered = m.allItems
		m.cursor = 0
		return
	}

	var filtered []Item
	for _, item := range m.allItems {
		if strings.Contains(strings.ToLower(item.FilterValue()), query) {
			filtered = append(filtered, item)
		}
	}
	m.filtered = filtered
	if m.cursor >= len(m.filtered) {
		m.cursor = max(0, len(m.filtered)-1)
	}
}

// Action returns the user's selected action after the TUI exits.
func (m Model) Action() interface{} {
	return m.action
}

// Quitting returns true if the user quit without selecting.
func (m Model) Quitting() bool {
	return m.quitting && m.action == nil
}

		"q quit",

// cmd/tui.go
package cmd

import (
	"fmt"
	"os"

	tea "github.com/charmbracelet/bubbletea"
	"github.com/drolosoft/cmux-resurrect/internal/gallery"
	"github.com/drolosoft/cmux-resurrect/internal/orchestrate"
	"github.com/drolosoft/cmux-resurrect/internal/tui"
	"github.com/spf13/cobra"
)

var tuiCmd = &cobra.Command{
	Use:   "tui",
	Short: "Interactive workspace launcher",
	Long:  "Fuzzy-search TUI for browsing saved layouts and templates.\nNavigate with arrow keys, type to filter, Enter to restore.",
	Args:  cobra.NoArgs,
	RunE:  runTUI,
}

func init() {
	rootCmd.AddCommand(tuiCmd)
}

func runTUI(cmd *cobra.Command, args []string) error {
	store, err := newStore()
	if err != nil {
		return err
	}

	// Load layout items.
	metas, err := store.List()
	if err != nil {
		return err
	}
	layoutItems := tui.ItemsFromLayouts(metas)

	// Load template items.
	templates := gallery.List()
	templateItems := tui.ItemsFromTemplates(templates)

	// Create and run the TUI.
	m := tui.NewModel(layoutItems, templateItems)
	p := tea.NewProgram(m, tea.WithAltScreen())
	finalModel, err := p.Run()
	if err != nil {
		return fmt.Errorf("tui: %w", err)
	}

	fm := finalModel.(tui.Model)

	// Handle the selected action.
	switch action := fm.Action().(type) {
	case tui.SelectedAction:
		return handleTUISelect(action.Item)
	case tui.SaveAction:
		return handleTUISave()
	case tui.DeleteAction:
		return handleTUIDelete(action.Item)
	default:
		// User quit without selecting.
		return nil
	}
}

func handleTUISelect(item tui.Item) error {
	switch item.Kind {
	case tui.KindLayout:
		// Restore the selected layout.
		cl := newClient()
		store, err := newStore()
		if err != nil {
			return err
		}

		restorer := &orchestrate.Restorer{
			Client: cl,
			Store:  store,
			OnProgress: func(title string, panes int, err error) {
				t := padTitle(title)
				if err != nil {
					fmt.Fprintf(os.Stderr, "  %s  %s: %v\n", yellowStyle.Render("!"), t, err)
				} else {
					fmt.Fprintf(os.Stderr, "  %s  %s (%d panes)\n", greenStyle.Render("OK"), t, panes)
				}
			},
		}

		fmt.Fprintf(os.Stderr, "\n%s %s\n", yellowStyle.Render("➕ Restoring"), greenStyle.Render(item.Name))
		result, err := restorer.Restore(item.Name, false, orchestrate.RestoreModeAdd)
		if err != nil {
			return err
		}
		fmt.Fprintf(os.Stderr, "\n%s\n\n",
			greenStyle.Render(fmt.Sprintf("✅ Restored %d/%d workspaces", result.WorkspacesOK, result.WorkspacesTotal)))
		return nil

	case tui.KindTemplate:
		// Apply the selected template (delegate to template use).
		return runTemplateUse(nil, []string{item.Name})
	}
	return nil
}

func handleTUISave() error {
	cl := newClient()
	store, err := newStore()
	if err != nil {
		return err
	}
	saver := &orchestrate.Saver{Client: cl, Store: store}

	fmt.Fprintln(os.Stderr)
	fmt.Fprintf(os.Stderr, "%s\n", yellowStyle.Render("💾 Saving current layout as 'default'"))
	layout, err := saver.Save("default", "")
	if err != nil {
		return err
	}
	fmt.Fprintf(os.Stderr, "%s\n\n",
		greenStyle.Render(fmt.Sprintf("✅ Saved %d workspaces", len(layout.Workspaces))))
	return nil
}

func handleTUIDelete(item tui.Item) error {
	store, err := newStore()
	if err != nil {
		return err
	}
	if err := store.Delete(item.Name); err != nil {
		return err
	}
	fmt.Fprintf(os.Stderr, "\n%s  Deleted layout %q\n\n", greenStyle.Render("✓"), item.Name)
	return nil
}
